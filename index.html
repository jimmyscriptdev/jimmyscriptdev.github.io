Script 1: Server Uptime Populator

Description: This script populates Airtable with server information using the `uptime` command.


#!/bin/bash
# Script Name: airtable_uptime.sh
# Creation Date: 2024-10-20
# Author: Jimmy
# Description: Populates Airtable with server uptime
curl -s -X POST https://api.airtable.com/v0/$__DATABASE/$__TABLE \
-H "Authorization: Bearer $__APIKEY" \
-H "Content-Type: application/json" \
--data '{ "records": [ { "fields": { "Name": "$__HOST", "Notes": "$(uptime)" } } ] }'

Script 2: System Monitor

Description: Monitors system memory and CPU usage.


#!/bin/bash
# Script Name: system_monitor.sh
# Creation Date: 2024-10-15
# Author: Jimmy
# Description: Monitors system memory and CPU usage
while true; do
    echo "Memory Usage: $(free -m | awk 'NR==2{printf "%.2f", $3*100/$2 }')%"
    echo "CPU Load: $(top -bn1 | grep "Cpu(s)" | sed "s/.*, *\([0-9.]*\)%* id.*/\1/" | awk '{print 100 - $1}')%"
    sleep 5
done
